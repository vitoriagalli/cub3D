#include "cub3d.h"
#define K_W 0x0077

int ft_move(int keycode, t_vars *vars)
{
    t_point center;
    t_data img;

    center.x = vars->pos->x;
    center.y = vars->pos->y;
    center.color = vars->pos->color;
    (void)vars;
    if (keycode == K_W)
    {
        center.y -= 10;
		vars->pos->y = center.y;
        img.img = mlx_new_image(vars->mlx, 600, 480);
        img.addr = mlx_get_data_addr(img.img, &img.bits_per_pixel, &img.line_length, &img.endian);
        ft_circle(&img, center, 100);
        mlx_put_image_to_window(vars->mlx, vars->win, img.img, 0, 0);
    }

    return (0);
}

int main(void)
{
    t_vars vars;
    t_point center;
    t_data img;

    vars.mlx = mlx_init();
    vars.win = mlx_new_window(vars.mlx, 600, 480, "Hook");
    center.x = 300;
    center.y = 300;
    center.color = 0x00ff0000;
    vars.pos = &center;
    img.img = mlx_new_image(vars.mlx, 600, 480);
    img.addr = mlx_get_data_addr(img.img, &img.bits_per_pixel, &img.line_length, &img.endian);
    ft_circle(&img, center, 100);
    mlx_put_image_to_window(vars.mlx, vars.win, img.img, 0, 0);

    mlx_key_hook(vars.win, ft_move, &vars);

    mlx_loop(vars.mlx);
}
